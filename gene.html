    function drawRoundRect(c, x, y, width, height, radius) {

      c.beginPath();

      c.moveTo(x + radius, y);

      c.lineTo(x + width - radius, y);

      c.quadraticCurveTo(x + width, y, x + width, y + radius);

      c.lineTo(x + width, y + height - radius);

      c.quadraticCurveTo(x + width, y + height, x + width - radius, y + height);

      c.lineTo(x + radius, y + height);

      c.quadraticCurveTo(x, y + height, x, y + height - radius);

      c.lineTo(x, y + radius);

      c.quadraticCurveTo(x, y, x + radius, y);

      c.closePath();

    }


    // --- ニューラルネットワーククラス ---

    class NeuralNetwork {

      constructor(inputSize, hiddenSize, outputSize) {

        this.inputSize = inputSize;

        this.hiddenSize = hiddenSize;

        this.outputSize = outputSize;


        this.weightsIH = []; 

        this.weightsHO = []; 


        this.biasesH = []; 

        this.biasesO = []; 


        this.initRandom();

      }


      initRandom() {

        for (let i = 0; i < this.inputSize; i++) {

          this.weightsIH[i] = [];

          for (let j = 0; j < this.hiddenSize; j++) {

            this.weightsIH[i][j] = Math.random() * 2 - 1;

          }

        }

        for (let i = 0; i < this.hiddenSize; i++) {

          this.weightsHO[i] = [];

          for (let j = 0; j < this.outputSize; j++) {

            this.weightsHO[i][j] = Math.random() * 2 - 1;

          }

        }

        for (let j = 0; j < this.hiddenSize; j++) {

          this.biasesH[j] = Math.random() * 2 - 1;

        }

        for (let j = 0; j < this.outputSize; j++) {

          this.biasesO[j] = Math.random() * 2 - 1;

        }

      }


      activate(x) {

        return 1 / (1 + Math.exp(-x));

      }


      feedForward(inputs) {

        const hiddenOutputs = [];

        for (let j = 0; j < this.hiddenSize; j++) {

          let sum = 0;

          for (let i = 0; i < this.inputSize; i++) {

            sum += inputs[i] * this.weightsIH[i][j];

          }

          sum += this.biasesH[j];

          hiddenOutputs[j] = this.activate(sum);

        }


        const outputs = [];

        for (let j = 0; j < this.outputSize; j++) {

          let sum = 0;

          for (let i = 0; i < this.hiddenSize; i++) {

            sum += hiddenOutputs[i] * this.weightsHO[i][j];

          }

          sum += this.biasesO[j];

          outputs[j] = this.activate(sum);

        }


        return { outputs, hiddenOutputs };

      }


      getGenome() {

        const genes = [];

        for (let i = 0; i < this.inputSize; i++) {

          genes.push(...this.weightsIH[i]);

        }

        for (let i = 0; i < this.hiddenSize; i++) {

          genes.push(...this.weightsHO[i]);

        }

        genes.push(...this.biasesH);

        genes.push(...this.biasesO);

        return genes;

      }


      loadGenome(genes) {

        let idx = 0;

        for (let i = 0; i < this.inputSize; i++) {

          for (let j = 0; j < this.hiddenSize; j++) {

            this.weightsIH[i][j] = genes[idx++];

          }

        }

        for (let i = 0; i < this.hiddenSize; i++) {

          for (let j = 0; j < this.outputSize; j++) {

            this.weightsHO[i][j] = genes[idx++];

          }

        }

        for (let j = 0; j < this.hiddenSize; j++) {

          this.biasesH[j] = genes[idx++];

        }

        for (let j = 0; j < this.outputSize; j++) {

          this.biasesO[j] = genes[idx++];

        }

      }

    }

          const yEnd = this.y + Math.sin(sAngle) * this.sensorRange;


          let closestDistance = this.sensorRange;


          walls.forEach(w => {

            const hit = getIntersection(

              { x: this.x, y: this.y },

              { x: xEnd, y: yEnd },

              { x: w.x1, y: w.y1 },

              { x: w.x2, y: w.y2 }

            );


            if (hit) {

              const d = Math.sqrt((hit.x - this.x)**2 + (hit.y - this.y)**2);

              if (d < closestDistance) {

                closestDistance = d;

              }

            }

          });


          this.sensors[i] = 1 - (closestDistance / this.sensorRange);

        }

      }


      update(walls) {

        if (!this.alive) return;


        this.totalFramesAlive++;

        this.timeSinceLastCheckpoint++;


        // 3秒チェック

        if (this.timeSinceLastCheckpoint > 180) {

          this.alive = false;

          sound.playCrash();

          return;

        }


        this.scan(walls);


        const { outputs, hiddenOutputs } = this.brain.feedForward(this.sensors);

        this.lastHiddenState = hiddenOutputs;

        this.lastOutputs = outputs;


        const thrust = (outputs[0] - 0.2) * 5; 

        const steering = (outputs[1] - 0.5) * 0.12; 


        this.angle += steering;

        this.speed = Math.max(0.5, Math.min(thrust, 4.5)); 


        this.x += Math.cos(this.angle) * this.speed;

        this.y += Math.sin(this.angle) * this.speed;


        if (this.checkCollision(walls)) {

          this.alive = false;

          sound.playCrash();

          return;

        }


        this.evalFitness();

      }


      checkCollision(walls) {

        const radius = 9;

        for (let i = 0; i < walls.length; i++) {

          const w = walls[i];

          const dist = distToSegment({ x: this.x, y: this.y }, { x: w.x1, y: w.y1 }, { x: w.x2, y: w.y2 });

          if (dist < radius) return true;

        }

        return false;

      }


      evalFitness() {

        for (let i = 0; i < checkLines.length; i++) {

          const targetIdx = (this.lastCheckpointIndex + 1) % checkLines.length;

          const line = checkLines[targetIdx];


          const dist = distToSegment({ x: this.x, y: this.y }, { x: line.x1, y: line.y1 }, { x: line.x2, y: line.y2 });

          if (dist < 18) {

            this.lastCheckpointIndex = targetIdx;

            this.checkpointsCleared++;

            this.timeSinceLastCheckpoint = 0; 

          }

        }


        this.fitness = (this.checkpointsCleared * 500) + this.totalFramesAlive;

      }


      draw(ctx) {

        ctx.save();

        ctx.translate(this.x, this.y);

        ctx.rotate(this.angle);


        if (this.isElite) {

          ctx.shadowBlur = 15;

          ctx.shadowColor = '#fbbf24'; 

          ctx.fillStyle = '#fbbf24';

        } else {

          ctx.shadowBlur = 8;

          ctx.shadowColor = this.color;

          ctx.fillStyle = this.color;

        }


        ctx.beginPath();

        ctx.moveTo(12, 0);

        ctx.lineTo(-8, -7);

        ctx.lineTo(-5, 0);

        ctx.lineTo(-8, 7);

        ctx.closePath();

        ctx.fill();


        ctx.restore();


        if (this.isElite && this.alive) {

          ctx.save();

          ctx.lineWidth = 1;

          for (let i = 0; i < this.sensorAngles.length; i++) {

            const sAngle = this.angle + this.sensorAngles[i];

            const sensorVal = this.sensors[i]; 

            const actLen = this.sensorRange * (1 - sensorVal);


            ctx.strokeStyle = sensorVal > 0.8 ? 'rgba(239, 68, 68, 0.4)' : 'rgba(16, 185, 129, 0.2)';

            ctx.beginPath();

            ctx.moveTo(this.x, this.y);

            ctx.lineTo(this.x + Math.cos(sAngle) * actLen, this.y + Math.sin(sAngle) * actLen);

            ctx.stroke();

          }

          ctx.restore();

        }

      }

    }

    class GeneticAlgorithm {

      constructor() {

        this.generation = 1;

        this.population = [];

        this.initPopulation();

      }


      initPopulation() {

        this.population = [];

        for (let i = 0; i < populationSize; i++) {

          this.population.push(new Car());

        }

        this.generation = 1;

        uiGeneration.innerText = this.generation;

      }


      evolve() {

        let maxFitness = -1;

        let bestCar = null;


        this.population.forEach(car => {

          if (car.fitness > maxFitness) {

            maxFitness = car.fitness;

            bestCar = car;

          }

        });


        uiBestFitness.innerText = (maxFitness / 100).toFixed(1);


        const nextGeneration = [];


        if (bestCar) {

          const elite = new Car();

          elite.brain.loadGenome(bestCar.brain.getGenome());

          elite.color = '#fbbf24'; 

          elite.isElite = true;

          nextGeneration.push(elite);

        }


        while (nextGeneration.length < populationSize) {

          const parentA = this.rouletteSelect();

          const parentB = this.rouletteSelect();


          const child = new Car();

          const childGenome = this.crossover(parentA.brain.getGenome(), parentB.brain.getGenome());

          this.mutate(childGenome);


          child.brain.loadGenome(childGenome);

          nextGeneration.push(child);

        }


        this.population = nextGeneration;

        this.generation++;

        uiGeneration.innerText = this.generation;


        sound.playGen();

      }


      rouletteSelect() {

        let totalFitness = 0;

        this.population.forEach(car => {

          totalFitness += Math.max(1, car.fitness);

        });


        let r = Math.random() * totalFitness;

        let sum = 0;

        for (let i = 0; i < this.population.length; i++) {

          sum += Math.max(1, this.population[i].fitness);

          if (sum >= r) {

            return this.population[i];

          }

        }

        return this.population[0];

      }


      crossover(parentAGenome, parentBGenome) {

        const childGenome = [];

        for (let i = 0; i < parentAGenome.length; i++) {

          childGenome[i] = Math.random() < 0.5 ? parentAGenome[i] : parentBGenome[i];

        }

        return childGenome;

      }


      mutate(genome) {

        for (let i = 0; i < genome.length; i++) {

          if (Math.random() < mutationRate) {

            genome[i] += (Math.random() * 0.4 - 0.2);

            if (genome[i] < -1) genome[i] = -1;

            if (genome[i] > 1) genome[i] = 1;

          }

        }

      }

    }

    function drawBrainStructure(eliteCar) {

      if (!eliteCar) return;


      const bw = brainCanvas.width / (window.devicePixelRatio || 1);

      const bh = brainCanvas.height / (window.devicePixelRatio || 1);

      bCtx.clearRect(0, 0, bw, bh);


      const net = eliteCar.brain;

      const neurons = [net.inputSize, net.hiddenSize, net.outputSize];

      

      const layerX = [40, bw / 2, bw - 40];

      const nodeY = [];


      for (let layerIdx = 0; layerIdx < neurons.length; layerIdx++) {

        nodeY[layerIdx] = [];

        const count = neurons[layerIdx];

        const step = bh / (count + 1);

        for (let nodeIdx = 0; nodeIdx < count; nodeIdx++) {

          nodeY[layerIdx][nodeIdx] = step * (nodeIdx + 1);

        }

      }


      for (let i = 0; i < net.inputSize; i++) {

        for (let h = 0; h < net.hiddenSize; h++) {

          const weight = net.weightsIH[i][h];

          const activeVal = eliteCar.sensors[i];


          bCtx.strokeStyle = weight > 0 ? `rgba(16, 185, 129, ${0.1 + activeVal * 0.5})` : `rgba(239, 68, 68, ${0.1 + activeVal * 0.5})`;

          bCtx.lineWidth = Math.abs(weight) * 3 * activeVal;

          

          bCtx.beginPath();

          bCtx.moveTo(layerX[0], nodeY[0][i]);

          bCtx.lineTo(layerX[1], nodeY[1][h]);

          bCtx.stroke();

        }

      }


      for (let h = 0; h < net.hiddenSize; h++) {

        for (let o = 0; o < net.outputSize; o++) {

          const weight = net.weightsHO[h][o];

          const activeVal = eliteCar.lastHiddenState[h] || 0;


          bCtx.strokeStyle = weight > 0 ? `rgba(16, 185, 129, ${0.1 + activeVal * 0.5})` : `rgba(239, 68, 68, ${0.1 + activeVal * 0.5})`;

          bCtx.lineWidth = Math.abs(weight) * 3 * activeVal;

          

          bCtx.beginPath();

          bCtx.moveTo(layerX[1], nodeY[1][h]);

          bCtx.lineTo(layerX[2], nodeY[2][o]);

          bCtx.stroke();

        }

      }


      for (let l = 0; l < neurons.length; l++) {

        const count = neurons[l];

        for (let n = 0; n < count; n++) {

          const x = layerX[l];

          const y = nodeY[l][n];


          let nodeVal = 0;

          if (l === 0) nodeVal = eliteCar.sensors[n];

          else if (l === 1) nodeVal = eliteCar.lastHiddenState[n] || 0;

          else nodeVal = eliteCar.lastOutputs[n] || 0;


          bCtx.save();

          bCtx.shadowBlur = 10;

          bCtx.shadowColor = '#00f0ff';

          bCtx.fillStyle = `rgba(0, 240, 255, ${0.3 + nodeVal * 0.7})`;

          bCtx.strokeStyle = '#ffffff';

          bCtx.lineWidth = 1.5;


          bCtx.beginPath();

          bCtx.arc(x, y, 7, 0, Math.PI * 2);

          bCtx.fill();

          bCtx.stroke();

          bCtx.restore();

        }

      }

    }

    // --- 📐 幾何計算ヘルパー ---


    function getIntersection(A, B, C, D) {

      const tTop = (D.x - C.x) * (A.y - C.y) - (D.y - C.y) * (A.x - C.x);

      const uTop = (C.y - A.y) * (A.x - B.x) - (C.x - A.x) * (A.y - B.y);

      const bottom = (D.y - C.y) * (B.x - A.x) - (D.x - C.x) * (B.y - A.y);


      if (bottom !== 0) {

        const t = tTop / bottom;

        const u = uTop / bottom;

        if (t >= 0 && t <= 1 && u >= 0 && u <= 1) {

          return {

            x: lerp(A.x, B.x, t),

            y: lerp(A.y, B.y, t),

            offset: t

          };

        }

      }

      return null;

    }


    function lerp(A, B, t) {

      return A + (B - A) * t;

    }


    function distToSegment(p, a, b) {

      const l2 = (a.x - b.x)**2 + (a.y - b.y)**2;

      if (l2 === 0) return Math.sqrt((p.x - a.x)**2 + (p.y - a.y)**2);

      let t = ((p.x - a.x) * (b.x - a.x) + (p.y - a.y) * (b.y - a.y)) / l2;

      t = Math.max(0, Math.min(1, t));

      return Math.sqrt((p.x - (a.x + t * (b.x - a.x)))**2 + (p.y - (a.y + t * (b.y - a.y)))**2);

    }
